 88:     def signup
 89:       @new_user = User.new
 90:       if request.post?
 91:         # If a non registered user has recieved an invitation but never accepted the invite, the user will already be in the database.
 92:         # If so, the user will still have a status of PENDING. Then, update the user instead of create a new user
 93:         # This is not secure since anyone could register in the name of someone else and get a hold of that persons invites
 94:         # For this to be secure we need to implement a confirmation my mail system for new signups.
 95:         @new_user = User.find_by_login_and_status(params[:user][:login], Status::PENDING) || User.new
 96:         @new_user.attributes = params[:user]
 97:         # do the user registration
 98:         @terms_of_service = params[:user][:terms_of_service]
 99:         @pwd_confirmation = params[:user][:pwd_confirmation]
100:         @login_confirmation = params[:user][:login_confirmation]
101:         @new_user.role_ids = [Role::USER] unless @new_user.in_role? Role::USER
102:         @new_user.status = Status::ACTIVE
103:         @new_user.created_by_id = current_user.id
104:         @new_user.updated_by_id = current_user.id
105:         @new_user.last_login_date = Time.now
106:         if @new_user.save
107:           logout
108:           self.current_user = @new_user
109:           create_welcome_board @new_user
110:           render_as_popup do
111:             render :layout => "decorator-public", :template => "public/how_stixy_works" and return
112:           end
113:         else
114:           render_as_popup do
115:             render :layout => "decorator-public", :template => "public/signup" and return
116:           end
117:         end
118:       else
119:         render_as_popup do
120:           render :layout => "decorator-public", :template => "public/signup" and return
121:         end
122:       end
