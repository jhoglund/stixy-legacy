63:   def clean_up widget_data
64:     # Clean-up widget data. There is a bug in the stixy.js file that results in a widget being submitet as
65:     # multiple instances in the xml post. When two or more xml nodes has the same name (<widget_nnn>)
66:     # rails combines the data of the nodes in an array and assigns the array as the value of a hash key named
67:     # after the node names. The fix below iterates through the array and megers all the members of the array into
68:     # a single hash. If there is conflicting key names, the later key will overwrite the former key, which could
69:     # leade to the lost of data. This bug has to be resolved on the client side (stixy.js) at a later stage. 
70:     # See bug ticket 35 (https://www.stixy.com/trac/ticket/35)
71:     if widget_data.instance_of?(Array)
72:       javascript_logger.server "Recieved multiple hashes for one widget: the data was #{widget_data.to_yaml}"
73:       tmp_hash = HashWithIndifferentAccess.new
74:       widget_data.each { |i| tmp_hash.merge!(i) }
75:       widget_data = tmp_hash
76:     end
77:     return widget_data
78:   end
